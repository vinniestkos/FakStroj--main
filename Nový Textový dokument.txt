 page.Margin(2, Unit.Centimetre);

 page.DefaultTextStyle(x => x.FontSize(20));

 page.Header()
     .Text("Hello PDF!");


 page.Content()
     .PaddingVertical(1, Unit.Centimetre)
     .Column(x =>
     {
         x.Spacing(20);

         x.Item().Text(Placeholders.LoremIpsum());
         x.Item().Image(Placeholders.Image(200, 100));
     });

 page.Footer()
     .AlignCenter()
     .Text(x =>
     {
         x.Span("Page ");
         x.CurrentPageNumber();
     });
